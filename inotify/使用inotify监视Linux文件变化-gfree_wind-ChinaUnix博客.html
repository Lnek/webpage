使用inotify监视Linux文件变化 2013-03-06 23:13:36

分类： LINUX

作者：gfree.wind@gmail.com
博客：blog.focus-linux.net   linuxfocus.blog.chinaunix.net 
微博：weibo.com/glinuxer
QQ技术群：4367710

Linux系统中提供了一套intotify的机制来监视文件系统的事件，比如创建文件，打开，关闭等等。利用这一机制，我们可以很容易写出监控目录变化的工具。更何况更有了一个inotify_tools的开源工程呢。inotify_tools对inotify的API做了进一步的封装，更加方便使用。

下面的例子来源于inotify_tools的示例代码，但是做了一些改进。inotify_tools监视一个目录时，可以得到该目录变化的信息。但是在该目录下，创建一个子目录后，子目录中的变化，inotify_tools不能获取。所以我添加了动态监控目录的代码。

  1. #include <stdlib.h>
  2. #include <stdio.h>

  4. #include <inotifytools/inotifytools.h>
  5. #include <inotifytools/inotify.h>

  7. enum {
  8.     ACTION_NULL_WD,
  9.     ACTION_ADD_WD,
  10.     ACTION_DEL_WD,
  11. };

  13. int main(int argc, const char **argv)
  14. {
  15.     int err = 0;

  17.     if (!inotifytools_initialize()) {
  18.         printf("inotifytools_initialize failedn");
  19.         goto error;
  20.     }

  22.     inotifytools_initialize_stats();

  24.     const char *monitor_path = ".";
  25.     if (argc > 1) {
  26.         monitor_path = argv[1];
  27.     }
  28.     printf("Monitor dir(%s)n", monitor_path);

  30.     if (!inotifytools_watch_recursively(monitor_path, IN_ALL_EVENTS)) {
  31.         printf("inotifytools_watch_recursively failedn");
  32.         goto error;
  33.     }

  35.     inotifytools_set_printf_timefmt("%F %T");

  37.     struct inotify_event * event = inotifytools_next_event(-1);
  38.     char path[256];
  39.     while (event) {
  40.         inotifytools_printf( event, "%T %w%f %en" );

  42.         if (IN_ISDIR&event->mask) {
  43.             int action = ACTION_NULL_WD;

  45.             if ((IN_DELETE|IN_DELETE_SELF|IN_MOVED_FROM)&event->mask) {
  46.                 action = ACTION_DEL_WD;
  47.                 snprintf(path, sizeof(path), "%s%s",
  48.                         inotifytools_filename_from_wd(event->wd),
  49.                         event->name);
  50.                 printf("Remove path(%s) from wdn", path);
  51.             } else if (((IN_CREATE|IN_MOVED_TO)&event->mask) && (IN_ISDIR&event->mask)) {
  52.                 action = ACTION_ADD_WD;
  53.                 snprintf(path, sizeof(path), "%s%s",
  54.                         inotifytools_filename_from_wd(event->wd),
  55.                         event->name);
  56.                 printf("Add path(%s) into wdn", path);
  57.             }

  59.             if (ACTION_ADD_WD == action) {
  60.                 if (!inotifytools_watch_recursively(path, IN_ALL_EVENTS)) {
  61.                     printf("inotifytools_watch_recursively failedn");
  62.                     goto error;
  63.                 }
  64.             } else if (ACTION_DEL_WD == action) {
  65.                 if (!inotifytools_remove_watch_by_wd(event->wd)) {
  66.                     printf("inotifytools_remove_watch_by_wd failed. event->wd(%d)n", event->wd);
  67.                     goto error;
  68.                 }
  69.             }
  70.         }
  71.         event = inotifytools_next_event(-1);
  72.     }

  74.     printf("Exitn");

  76.     return 0;

  78. error:
  79.     err = inotifytools_error();
  80.     printf("Error(%d)n", err);
  81.     return -1;
  82. }

代码很简单。就是在获得事件以后，检查是否为目录。如果是目录，则需要进行动态监控的检查。如果是创建动作，那么就添加新的监控。如果是删除动作，就去掉已有的监控wd。
